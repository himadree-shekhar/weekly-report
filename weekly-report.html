-- This query finds the article in PHPKB that has a custom field "CHM_HELP_ID" with the value 10225
SELECT
a.article_id,
a.article_title
FROM
phpkb_articles a
JOIN phpkb_custom_data d ON a.article_id = d.article_id -- Link custom data to article
JOIN phpkb_custom_fields f ON d.field_id = f.field_id -- Link custom data to custom field
WHERE
f.field_name = 'CHM_HELP_ID' -- Filter to only use CHM Help ID field
AND d.field_data = '10225'; -- Lookup CHM Help ID (from Delphi)
